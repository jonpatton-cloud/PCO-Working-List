  <script>
    async function loadData() {
      const sheetId = "1GdLcZO7AT91gb5s2zcUVCAcI20Fppk22q-DK7Roeb8k";
      const sheetName = "Sheet1"; // Change this if your tab has a different name
      const url = `https://docs.google.com/spreadsheets/d/${sheetId}/gviz/tq?tqx=out:json&sheet=${encodeURIComponent(sheetName)}`;

      try {
        const res = await fetch(url);
        const text = await res.text();
        const json = JSON.parse(text.match(/.*\((.*)\)/s)[1]);
        
        const cols = json.table.cols.map(c => c.label || "");
        const rows = json.table.rows;

        let html = "<table><thead><tr>";
        cols.forEach(label => { html += `<th>${label}</th>`; });
        html += "</tr></thead><tbody>";
        rows.forEach(r => {
          html += "<tr>";
          r.c.forEach(cell => html += `<td>${cell ? cell.v : ""}</td>`);
          html += "</tr>";
        });
        html += "</tbody></table>";
        document.getElementById("data").innerHTML = html;
      } catch (err) {
        document.getElementById("data").innerHTML = `<div class="error">Error: ${err}</div>`;
        console.error(err);
      }
    }
    window.onload = loadData;
  </script>
